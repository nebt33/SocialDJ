server protocol sketch
server->client messages:
	song database:
		download_starting|new download_id
		download_fail|delete download_id
		download_success|delete download_id|new song_id
		new_song|new song_id
		song_info|song_id|metaitem|value
		forget_song|delete song_id
		new_album|new album_id
		album_info|album_id|name|song_ids
		forget_album|delete album_id
		new_artist|new artist_id
		artist_info|artist_id|name
		forget_artist|delete artist_id
	queue:
		add_bottom|song_id
		remove_top|song_id
		score|song_id|value
	playback:
		playing
		paused
		skip
client->server messages:
	song database:
		download_song|url
		list_songs|[|metaitem|value]|start|count
		list_albums|[|metaitem|value]|start|count
		list_artists|[|metaitem|value]|start|count
		delete_song|song_id
	queue:
		queue_add|song_id
	vote|song_id|value
	playback:
	play
	pause
	skip
metaitems are one of: artist, album, title, duration
